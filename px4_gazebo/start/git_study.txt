git init        (初始化)
git status      (查看状态)
git add .       (添加到缓冲区)
git commit -m "完成基础仿真"        (提交)    
git tag v1.0.0      （添加版本标签）
git tag -d v1       （删除某个版本标签，此处删除v1）
git log --oneline       （查看）


# 1. 关联远程地址（把 https://... 换成你 Gitee 上实际的地址）
git remote add origin https://gitee.com/你的用户名/px4_gazebo.git
（如果填写错误）
git remote set-url origin https://gitee.com/你的用户名/px4_gazebo.git

# 2. 把本地代码推上去
git push -u origin master

# 3. 把标签也推上去
git push origin --tags


# 查看当前远程地址
git remote -v

# 删除错误的 origin
git remote remove origin

# 重新添加正确的
git remote add origin https://gitee.com/W3378516198/px4_gazebo.git
